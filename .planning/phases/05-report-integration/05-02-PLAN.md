class Pipeline:
    """Orchestrates the full JailbreakForge attack pipeline."""
    
    def __init__(
        self,
        config: AppConfig,
        attacker: AttackerAgent,
        target: TargetConnector,
        judge: JudgeAgent,
        db: Database,
        reporter: ReportGenerator,
    ):
        self.config = config
        self.attacker = attacker
        self.target = target
        self.judge = judge
        self.db = db
        self.reporter = reporter
    
    def execute(self, run_config: RunConfig) -> RunSummary:
        """Execute the full attack pipeline."""
        
        # 1. Create run in DB
        self.db.create_run(run_config)
        
        # 2. Load seeds (filtered by categories if specified)
        seeds = load_seeds(run_config.categories if run_config.categories != ["all"] else None)
        
        results: list[AttackResult] = []
        total_seeds = sum(len(s) for s in seeds.values())
        
        # 3. Rich progress bar
        with Progress(
            SpinnerColumn(),
            TextColumn("[progress.description]{task.description}"),
            BarColumn(),
            TaskProgressColumn(),
            TimeElapsedColumn(),
            console=console,
        ) as progress:
            task = progress.add_task("🔥 Running attacks...", total=total_seeds)
            
            # 4. For each category and seed:
            for category, seed_list in seeds.items():
                for seed in seed_list:
                    result = self._attack_seed(seed, run_config, progress, task)
                    results.append(result)
                    self.db.save_attack(run_config.run_id, result)
                    progress.advance(task)
        
        # 5. Calculate summary stats
        summary = self._build_summary(run_config, results)
        
        # 6. Update run in DB
        self.db.complete_run(run_config.run_id, summary)
        
        # 7. Generate report
        report_path = self.reporter.generate(summary)
        
        return summary, report_path
    
    def _attack_seed(self, seed: SeedTemplate, config: RunConfig, progress, task) -> AttackResult:
        """Execute attack for a single seed with mutation retries."""
        target_desc = f"{config.target_name} ({config.target_model})"
        
        # Generate initial attack
        attack = self.attacker.generate(seed, target_desc)
        
        for attempt in range(config.max_mutations + 1):
            # Send to target
            progress.update(task, description=f"⚔️ [{seed.category}] {seed.name} (attempt {attempt+1})")
            
            try:
                response = self.target.send_prompt(attack.final_prompt)
            except Exception as e:
                response = f"[ERROR] Target returned: {str(e)}"
            
            # Judge the response
            verdict = self.judge.evaluate(response, attack.category)
            
            # If blocked and we have mutations left, mutate
            if verdict.verdict == "BLOCKED" and attempt < config.max_mutations:
                attack = self.attacker.mutate(attack, verdict.reasoning, seed.mutation_hints)
                continue
            
            # Return result (either successful attack, final mutation, or last blocked)
            return AttackResult(
                attack=attack,
                target_response=response,
                verdict=verdict,
                timestamp=datetime.now().isoformat()
            )
    
    def _build_summary(self, config: RunConfig, results: list[AttackResult]) -> RunSummary:
        """Build RunSummary from results."""
        total = len(results)
        blocked = sum(1 for r in results if r.verdict.verdict == "BLOCKED")
        vulnerable = sum(1 for r in results if r.verdict.verdict == "VULNERABLE")
        partial = sum(1 for r in results if r.verdict.verdict == "PARTIAL")
        score = ReportGenerator.calculate_security_score(total, blocked)
        
        return RunSummary(
            config=config,
            results=results,
            total_attacks=total,
            total_blocked=blocked,
            total_vulnerable=vulnerable,
            total_partial=partial,
            security_score=score,
            completed_at=datetime.now().isoformat()
        )
```

@app.command()
def run(
    target: str = typer.Option(..., "--target", "-t", help="Target API endpoint URL"),
    model: str = typer.Option(..., "--model", "-m", help="Target model name"),
    target_api_key: str = typer.Option(..., "--target-key", "-k", help="Target API key"),
    target_type: str = typer.Option("openai", "--type", help="Target type: openai, anthropic, generic"),
    categories: Optional[str] = typer.Option(None, "--categories", "-c", help="Comma-separated categories"),
    max_mutations: int = typer.Option(3, "--max-mutations", help="Max mutations per seed (1-5)"),
    provider: str = typer.Option("groq", "--provider", "-p", help="LLM provider: groq, openrouter"),
    output_dir: str = typer.Option("reports", "--output", "-o", help="Output directory"),
    target_name: str = typer.Option("target", "--name", "-n", help="Human-friendly target name"),
):
    """🔒 Run a security audit against a target AI model."""
    print_banner()
    print_disclaimer()
    
    # Load config
    config = load_config()
    
    # Parse categories
    cat_list = categories.split(",") if categories else None
    
    # Create RunConfig
    run_config = RunConfig(
        run_id=str(uuid4()),
        target_name=target_name,
        target_endpoint=target,
        target_model=model,
        categories=cat_list or list_categories(),
        max_mutations=max_mutations,
        provider=provider,
        attacker_model=config.attacker_model,
        judge_model=config.judge_model,
        started_at=datetime.now().isoformat()
    )
    
    # Create components
    api_key = get_active_api_key(config)
    attacker_provider = create_provider(provider, api_key, config.attacker_model)
    judge_provider = create_provider(provider, api_key, config.judge_model)
    target_conn = create_target(target_type, target, model, target_api_key)
    
    attacker = AttackerAgent(attacker_provider)
    judge = JudgeAgent(judge_provider)
    db = Database()
    reporter = ReportGenerator(output_dir)
    
    # Run pipeline
    pipeline = Pipeline(config, attacker, target_conn, judge, db, reporter)
    
    with spinner("Initializing audit..."):
        pass  # Components already created
    
    summary, report_path = pipeline.execute(run_config)
    
    # Print results
    rating, emoji = ReportGenerator.get_rating(summary.security_score)
    console.print(f"\n{emoji} Security Score: {summary.security_score:.1f}/100 ({rating})")
    console.print(f"📊 Attacks: {summary.total_attacks} | 🟢 Blocked: {summary.total_blocked} | 🔴 Vulnerable: {summary.total_vulnerable} | 🟡 Partial: {summary.total_partial}")
    console.print(f"📄 Report: {report_path}")
    
    db.close()
```
