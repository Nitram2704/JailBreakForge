```python
class AnthropicTarget(TargetConnector):
    """Connector for Anthropic Messages API."""
    
    def send_prompt(self, prompt: str) -> str:
        headers = {
            "x-api-key": self.api_key,
            "anthropic-version": "2023-06-01",
            "Content-Type": "application/json"
        }
        payload = {
            "model": self.model,
            "max_tokens": 1024,
            "messages": [{"role": "user", "content": prompt}]
        }
        
        response = self._retry_with_backoff(
            lambda: self.client.post(
                f"{self.endpoint}/v1/messages",
                headers=headers,
                json=payload
            )
        )
        
        data = response.json()
        # Anthropic returns content as a list of blocks
        return data["content"][0]["text"]
```

```python
def create_target(target_type: str, endpoint: str, model: str, api_key: str, **kwargs) -> TargetConnector:
    targets = {
        "openai": OpenAITarget,
        "anthropic": AnthropicTarget,
        "generic": GenericTarget,
    }
    if target_type not in targets:
        raise ValueError(f"Unknown target type: {target_type}")
    return targets[target_type](endpoint=endpoint, model=model, api_key=api_key, **kwargs)
```
