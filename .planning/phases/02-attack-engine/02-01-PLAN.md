```python
@dataclass
class SeedTemplate:
    category: str
    name: str
    description: str
    base_prompt: str
    mutation_hints: list[str]

def load_seeds(categories: Optional[list[str]] = None) -> dict[str, list[SeedTemplate]]:
    """Load seed templates from YAML files in the seeds/ directory.
    
    Args:
        categories: Optional filter — only load these categories. None = all.
    
    Returns:
        Dict mapping category name to list of SeedTemplate objects.
    """
    seeds_dir = Path(__file__).parent / "seeds"
    result = {}
    for yaml_file in sorted(seeds_dir.glob("*.yaml")):
        data = yaml.safe_load(yaml_file.read_text(encoding="utf-8"))
        category = data["category"]
        if categories and category not in categories:
            continue
        templates = [
            SeedTemplate(
                category=category,
                name=s["name"],
                description=s["description"],
                base_prompt=s["base_prompt"],
                mutation_hints=s["mutation_hints"]
            )
            for s in data["seeds"]
        ]
        result[category] = templates
    return result

def list_categories() -> list[str]:
    """Return sorted list of available attack category names."""
    seeds = load_seeds()
    return sorted(seeds.keys())
```
